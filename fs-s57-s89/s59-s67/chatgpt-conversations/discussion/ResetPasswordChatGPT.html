Generate an Express.js code snippet that includes a route and a controller to implement a password reset functionality for users. Here are the requirements: We will pass a JWT token containing the user ID in the authorization headers. The request body should only contain the new password. The app uses bcrypt and JWT for password hashing and token generation. The app uses the Mongoose library for interacting with the MongoDB database. The app already has register and login routes implemented. The app follows the Model-Controller-Route (MCR) architecture. Make sure to separate the route and controller for the password reset functionality.

Sure! Here's an example of an Express.js code snippet that includes a route and a controller to implement password reset functionality for users, following the mentioned requirements:

In the routes/resetPassword.js file:

javascript
const express = require('express');
const router = express.Router();
const resetPasswordController = require('../controllers/resetPasswordController');
const authMiddleware = require('../middlewares/authMiddleware');

// POST route for resetting the password
router.post('/reset-password', authMiddleware, resetPasswordController.resetPassword);

module.exports = router;

In the controllers/resetPasswordController.js file:

javascript
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const User = require('../models/User');

// Function to reset the password
const resetPassword = async (req, res) => {
  try {
    const { newPassword } = req.body;
    const { userId } = req.user; // Extracting user ID from the authorization header

    // Hashing the new password
    const hashedPassword = await bcrypt.hash(newPassword, 10);

    // Updating the user's password in the database
    await User.findByIdAndUpdate(userId, { password: hashedPassword });

    // Sending a success response
    res.status(200).json({ message: 'Password reset successfully' });
  } catch (error) {
    console.error(error);
    res.status(500).json({ message: 'Internal server error' });
  }
};

module.exports = {
  resetPassword
};

In the middlewares/authMiddleware.js file (to validate the JWT token and extract the user ID):

javascript
const jwt = require('jsonwebtoken');

// Middleware function to verify the JWT token and extract the user ID
const authMiddleware = (req, res, next) => {
  const token = req.headers.authorization;

  // Checking if the token is provided
  if (!token) {
    return res.status(401).json({ message: 'Unauthorized' });
  }

  try {
    // Verifying the token and extracting the user ID
    const decoded = jwt.verify(token, 'your-secret-key');
    req.user = { userId: decoded.userId };
    next();
  } catch (error) {
    console.error(error);
    res.status(401).json({ message: 'Invalid token' });
  }
};

module.exports = authMiddleware;

Please note that you need to replace 'your-secret-key' in the authMiddleware.js file with your actual secret key used for JWT token verification.

Make sure to install the required dependencies (e.g., Express, bcrypt, jwt, mongoose) and configure the database connection in your main application file.

This code snippet follows the Model-Controller-Route (MCR) architecture, with separate route and controller files for the password reset functionality.
