ExpressJS Course Search

June 30, 2023
Generate an expressjs code snippet of a route and a controller to implement a search functionality for courses by course name. Allow users to search for courses based on course name. The course name is passed in the request body. Assume that expressjs app uses mongoose Assume that the app uses Models, Controllers, Route Architecture.

Sure! Here's an example of an Express.js code snippet that implements a search functionality for courses based on course name using Mongoose, with a Model-Controller-Route architecture:

course.model.js:

javascript
const mongoose = require('mongoose');

const courseSchema = new mongoose.Schema({
  name: {
    type: String,
    required: true
  },
  // Other course properties...
});

const Course = mongoose.model('Course', courseSchema);

module.exports = Course;

course.controller.js:

javascript
const Course = require('./course.model');

// Controller action to search for courses by course name
const searchCoursesByName = async (req, res) => {
  try {
    const { courseName } = req.body;

    // Use a regular expression to perform a case-insensitive search
    const courses = await Course.find({
      name: { $regex: courseName, $options: 'i' }
    });

    res.json(courses);
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Internal Server Error' });
  }
};

module.exports = {
  searchCoursesByName
};

course.route.js:

javascript
const express = require('express');
const courseController = require('./course.controller');

const router = express.Router();

// Route to search for courses by course name
router.post('/courses/search', courseController.searchCoursesByName);

module.exports = router;

app.js:

javascript
const express = require('express');
const mongoose = require('mongoose');
const courseRoutes = require('./course.route');

const app = express();

// Connect to MongoDB using Mongoose
mongoose.connect('mongodb://localhost:27017/myapp', {
  useNewUrlParser: true,
  useUnifiedTopology: true,
  useCreateIndex: true
})
  .then(() => {
    console.log('Connected to MongoDB');
  })
  .catch(error => {
    console.error('Failed to connect to MongoDB', error);
  });

// Middleware to parse request bodies as JSON
app.use(express.json());

// Route middleware
app.use(courseRoutes);

// Start the server
app.listen(3000, () => {
  console.log('Server started on port 3000');
});

Make sure to install the necessary dependencies such as express, mongoose, and any other required packages via npm or yarn before running this code.

In this example, when a POST request is sent to /courses/search, the course name is expected to be in the request body. The controller action searchCoursesByName uses a case-insensitive regular expression to search for courses whose name matches the provided search term. The resulting courses are returned as a JSON response.
